import chess
from pyscript import document
from pyodide.ffi import create_proxy

# Initialize the board state
board = chess.Board()

def render_board():
    board_element = document.querySelector("#board-output")
    # board.unicode() gives a clean text representation of the board
    board_element.innerText = str(board.unicode(invert_color=True))

def process_move(event):
    # Only trigger when the user presses the Enter key
    if event.key == "Enter":
        input_element = document.querySelector("#move-input")
        error_element = document.querySelector("#error-msg")
        move_str = input_element.value.strip()
        
        # Clear inputs/errors immediately
        input_element.value = ""
        error_element.innerText = ""
        
        if not move_str:
            return

        try:
            # Parse UCIS move (e.g., e2e4)
            move = chess.Move.from_uci(move_str)
            
            if move in board.legal_moves:
                board.push(move)
                render_board()
                
                # Check for game over states
                if board.is_game_over():
                    error_element.innerText = f"Game Over. Result: {board.result()}"
            else:
                error_element.innerText = "Illegal move. Try again."
        except ValueError:
            error_element.innerText = "Invalid format. Use UCI format (e.g., e2e4)."

# Initial board load
render_board()

# Bind the python keydown function to the HTML input field
input_proxy = create_proxy(process_move)
document.querySelector("#move-input").addEventListener("keydown", input_proxy)
